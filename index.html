心跳保活

POST/api/heartbeat

请求体 JSON：key · device_id · appid

建议每 60–90 秒调用一次以维持在线状态（判定窗口 2 分钟）。可与签名字段配合使用。

{ "code": 200, "alive": true, "status": "active",
  "data": { "type": "days", "remaining": 25,
            "remainingMs": 2160000000, "deviceId": "my-device" }}

HMAC 签名说明

当应用开启 需要签名验证 时，所有请求必须携带有效的 HMAC-SHA256 签名。

// 构造签名载荷
payload   = "{appid}:{key}:{timestamp}:{nonce}"
signature = HMAC-SHA256(appSecret, payload).hexdigest()

// 请求时附加字段（GET 请求放查询参数，POST 请求放请求体）
_appid     = 你的应用 ID
_timestamp = Unix 秒级时间戳（必须在服务器时间 ±300 秒内）
_nonce     = 随机十六进制字符串（每次请求唯一，防重放攻击）
_sign      = 计算得到的签名

代码示例

import requests, hashlib, hmac, time, uuid

BASE   = "http://localhost:3000"   # 替换为你的服务器地址
APPID  = "demo-app"
SECRET = "your-app-secret"         # 从 管理后台 > 应用管理 获取

def sign(key):
    ts = str(int(time.time()))
    nc = uuid.uuid4().hex[:16]
    payload = f"{APPID}:{key}:{ts}:{nc}"
    sig = hmac.new(SECRET.encode(), payload.encode(), hashlib.sha256).hexdigest()
    return {"_appid":APPID, "_timestamp":ts, "_nonce":nc, "_sign":sig}

def verify(key, device_id=""):
    r = requests.get(f"{BASE}/api/verify", params={"key":key, "device_id":device_id, **sign(key)})
    return r.json()

def activate(key, device_id=""):
    r = requests.post(f"{BASE}/api/activate", json={"key":key, "device_id":device_id, **sign(key)})
    return r.json()

# 使用示例
result = verify("KM-XXXX-XXXX-XXXX-XXXX")
if result["valid"]:
    print("卡密有效，状态：", result["status"])
using System.Net.Http;
using System.Security.Cryptography;
using System.Text;
using System.Text.Json;

public class CardKeyClient {
    private static readonly HttpClient Http = new();
    const string BASE   = "http://localhost:3000";  // 替换为你的服务器地址
    const string APPID  = "demo-app";
    const string SECRET = "your-app-secret";  // 从 管理后台 > 应用管理 获取

    static Dictionary<string,string> Sign(string key) {
        var ts = DateTimeOffset.UtcNow.ToUnixTimeSeconds().ToString();
        var nc = Guid.NewGuid().ToString("N")[..16];
        var payload = $"{APPID}:{key}:{ts}:{nc}";
        using var h = new HMACSHA256(Encoding.UTF8.GetBytes(SECRET));
        var sig = Convert.ToHexString(h.ComputeHash(Encoding.UTF8.GetBytes(payload))).ToLower();
        return new() { ["_appid"]=APPID, ["_timestamp"]=ts, ["_nonce"]=nc, ["_sign"]=sig };
    }

    public static async Task<JsonElement> Verify(string key, string deviceId = "") {
        var s = Sign(key);
        var url = $"{BASE}/api/verify?key={key}&device_id={deviceId}&_appid={s["_appid"]}&_timestamp={s["_timestamp"]}&_nonce={s["_nonce"]}&_sign={s["_sign"]}";
        return JsonDocument.Parse(await Http.GetStringAsync(url)).RootElement;
    }

    public static async Task<JsonElement> Activate(string key, string deviceId = "") {
        var body = new Dictionary<string,object>{ ["key"]=key, ["device_id"]=deviceId };
        foreach (var kv in Sign(key)) body[kv.Key] = kv.Value;
        var resp = await Http.PostAsync($"{BASE}/api/activate",
            new StringContent(JsonSerializer.Serialize(body), Encoding.UTF8, "application/json"));
        return JsonDocument.Parse(await resp.Content.ReadAsStringAsync()).RootElement;
    }
}
// Node.js（使用内置 crypto 模块）
const crypto = require('crypto');
const BASE   = 'http://localhost:3000';  // 替换为你的服务器地址
const APPID  = 'demo-app';
const SECRET = 'your-app-secret';  // 从 管理后台 > 应用管理 获取

function sign(key) {
  const ts = String(Math.floor(Date.now() / 1000));
  const nc = crypto.randomBytes(8).toString('hex');
  const payload = `${APPID}:${key}:${ts}:${nc}`;
  const sig = crypto.createHmac('sha256', SECRET).update(payload).digest('hex');
  return { _appid: APPID, _timestamp: ts, _nonce: nc, _sign: sig };
}

async function verify(key, deviceId = '') {
  const p = new URLSearchParams({ key, device_id: deviceId, ...sign(key) });
  return (await fetch(`${BASE}/api/verify?${p}`)).json();
}

async function activate(key, deviceId = '') {
  const r = await fetch(`${BASE}/api/activate`, {
    method: 'POST', headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ key, device_id: deviceId, ...sign(key) })
  });
  return r.json();
}
package main

import (
    "crypto/hmac"; "crypto/sha256"; "encoding/hex"; "encoding/json"
    "fmt"; "io"; "net/http"; "strings"; "time"; "crypto/rand"
)

const (
    BASE   = "http://localhost:3000"  // 替换为你的服务器地址
    APPID  = "demo-app"
    SECRET = "your-app-secret"  // 从 管理后台 > 应用管理 获取
)

func sign(key string) (ts, nc, sig string) {
    ts = fmt.Sprintf("%d", time.Now().Unix())
    b := make([]byte, 8); rand.Read(b); nc = hex.EncodeToString(b)
    payload := fmt.Sprintf("%s:%s:%s:%s", APPID, key, ts, nc)
    mac := hmac.New(sha256.New, []byte(SECRET))
    mac.Write([]byte(payload)); sig = hex.EncodeToString(mac.Sum(nil))
    return
}

func Verify(key, deviceId string) map[string]interface{} {
    ts, nc, sg := sign(key)
    url := fmt.Sprintf("%s/api/verify?key=%s&device_id=%s&_appid=%s&_timestamp=%s&_nonce=%s&_sign=%s",
        BASE, key, deviceId, APPID, ts, nc, sg)
    resp, _ := http.Get(url)
    defer resp.Body.Close()
    body, _ := io.ReadAll(resp.Body)
    var result map[string]interface{}; json.Unmarshal(body, &result)
    return result
}
